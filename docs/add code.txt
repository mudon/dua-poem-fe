    const reportSpan = container.querySelector(`.report-indicator[data-type="${type}"][data-id="${itemId}"] .report-badge-red`);
    if (reportSpan && reportCount > 0) reportSpan.classList.add('show');
    return { type, id: itemId, title };
  }

  // --- Report logic ---
  function updateBadgeForItem(type, itemId) {
    const selector = `.report-indicator[data-type="${type}"][data-id="${itemId}"] .report-badge-red`;
    const badgeSpan = document.querySelector(selector);
    if (!badgeSpan) return;
    const count = reportsArray.filter(r => r.type === type && r.itemId == itemId).length;
    badgeSpan.innerText = count;
    if (count > 0) badgeSpan.classList.add('show');
    else badgeSpan.classList.remove('show');
  }
  function updateAllBadges() {
    document.querySelectorAll('.report-indicator').forEach(ind => {
      const type = ind.getAttribute('data-type');
      const id = parseInt(ind.getAttribute('data-id'));
      if (type && id) updateBadgeForItem(type, id);
    });
  }
  function saveReports() { localStorage.setItem('reportsArray', JSON.stringify(reportsArray)); }
  function addReport(type, itemId, title, reason, description) {
    reportsArray.unshift({ id: nextReportId++, type, itemId, title, reason, description: description || 'No description', timestamp: new Date().toISOString() });
    saveReports();
    updateAllBadges();
  }
  function showReportPopout(type, itemId, title) {
    const itemReports = reportsArray.filter(r => r.type === type && r.itemId == itemId);
    document.getElementById('popoutItemTitle').innerText = title;
    const container = document.getElementById('popoutReportList');
    if (itemReports.length === 0) container.innerHTML = '<div style="padding:2rem; text-align:center; color:#B7AB98;">📭 No reports for this content.</div>';
    else {
      let html = '';
      itemReports.forEach(r => {
        const reasonText = r.reason.replace(/_/g, ' ').replace(/\b\w/g, l => l.toUpperCase());
        html += `<div class="report-item"><div><span class="report-reason">${reasonText}</span></div><div class="report-desc">📝 ${escapeHtml(r.description)}</div><div class="report-meta"><i class="far fa-clock"></i> ${new Date(r.timestamp).toLocaleString()}</div></div>`;
      });
      container.innerHTML = html;
    }
    document.getElementById('reportPopout').classList.add('open');
    document.getElementById('popoutOverlay').classList.add('active');
  }
  function closeReportPopout() {
    document.getElementById('reportPopout').classList.remove('open');
    document.getElementById('popoutOverlay').classList.remove('active');
  }

  // Detail page
  function findItem(type, id) {
    const collection = type === 'dua' ? allDuas : allPoems;
    return collection.find(i => i.id == id);
  }
  function showDetailPage(type, id) {
    const data = findItem(type, id);
    if (!data) return;
    let html = `<h2 style="margin-bottom:0.5rem;">${data.title}</h2><div><span class="verified ${!data.verified ? 'pending' : ''}">${data.verified ? '✓ Verified' : '⏳ Pending Review'}</span></div>`;
    if (type === 'dua') {
      if (data.arabic) html += `<div class="detail-arabic">${data.arabic}</div>`;
      if (data.transliteration) html += `<div class="detail-field"><div class="detail-label">Transliteration</div><div class="detail-text">${data.transliteration}</div></div>`;
      if (data.translation) html += `<div class="detail-field"><div class="detail-label">Translation</div><div class="detail-text">${data.translation}</div></div>`;
      if (data.category) html += `<div class="detail-field"><div class="detail-label">Category</div><div class="detail-text">${data.category}</div></div>`;
    } else {
      if (data.author) html += `<div class="detail-field"><div class="detail-label">Author</div><div class="detail-text">${data.author}</div></div>`;
      if (data.content) html += `<div class="detail-field"><div class="detail-label">Poem text</div><div class="detail-text">${data.content}</div></div>`;
      if (data.translation) html += `<div class="detail-field"><div class="detail-label">Translation</div><div class="detail-text">${data.translation}</div></div>`;
    }
    if (data.tags) html += `<div class="detail-field"><div class="detail-label">Tags</div><div>${data.tags.map(t => `<span class="tag">${t}</span>`).join('')}</div></div>`;
    html += `<button class="detail-report-btn" id="detailReportBtn"><i class="fas fa-flag"></i> Report this content</button>`;
    document.getElementById('detailContent').innerHTML = html;
    document.getElementById('homeSection').classList.remove('active-section');
    document.getElementById('myDuasSection').classList.remove('active-section');
    document.getElementById('myPoemsSection').classList.remove('active-section');
    document.getElementById('profilePage').classList.remove('active');
    document.getElementById('detailPage').classList.add('active');
    document.getElementById('detailReportBtn').addEventListener('click', () => {
      let reason = prompt('Report reason (wrong_translation, inappropriate, duplicate, other):', 'wrong_translation');
      if (!reason) return;
      let desc = prompt('Description (optional):', '');
      addReport(type, id, data.title, reason.trim().toLowerCase(), desc || 'No description');
      alert('Report submitted.');
    });
  }
  function hideDetailPage() {

  function escapeHtml(str) { return str.replace(/[&<>]/g, m => ({'&':'&amp;','<':'&lt;','>':'&gt;'})[m]); }

  // Event handlers
  function attachAllListeners() {
    // User info click -> profile
    document.querySelectorAll('.user-info').forEach(el => {
      el.removeEventListener('click', userClickHandler);
      el.addEventListener('click', userClickHandler);
    });
    document.querySelectorAll('.report-indicator').forEach(btn => {
      btn.removeEventListener('click', reportHandler);
      btn.addEventListener('click', reportHandler);
    });
    document.querySelectorAll('.bookmark-btn').forEach(btn => {
      btn.removeEventListener('click', bookmarkHandler);
      btn.addEventListener('click', bookmarkHandler);
    });
    document.querySelectorAll('.dua-card, .poem-card').forEach(card => {
      card.removeEventListener('click', cardClickHandler);
      card.addEventListener('click', cardClickHandler);
    });
    document.querySelectorAll('.like-btn').forEach(btn => {
      btn.removeEventListener('click', likeHandler);
      btn.addEventListener('click', likeHandler);
    });
  }
  function userClickHandler(e) {
    e.stopPropagation();
    const userName = this.getAttribute('data-user-name');
    if (userName) showProfile(userName);
  }
  function reportHandler(e) {
    e.stopPropagation();
    const btn = e.currentTarget;
    const type = btn.getAttribute('data-type');
    const id = parseInt(btn.getAttribute('data-id'));
    const title = btn.getAttribute('data-title');
    showReportPopout(type, id, title);
  }
  function bookmarkHandler(e) {
    e.stopPropagation();
    const btn = e.currentTarget;
    const type = btn.getAttribute('data-type');
    const id = parseInt(btn.getAttribute('data-id'));
    const icon = btn.querySelector('i');
    if (icon.classList.contains('far')) {
      incrementSave(type, id);
      icon.classList.replace('far', 'fas');
      btn.querySelector('.bookmark-count').innerText = getSaveCount(type, id);
      alert('Bookmarked');
    } else {
      decrementSave(type, id);
      icon.classList.replace('fas', 'far');
      btn.querySelector('.bookmark-count').innerText = getSaveCount(type, id);
      alert('Removed');
    }
  }
  function cardClickHandler(e) {
    if (e.target.closest('.report-indicator') || e.target.closest('.bookmark-btn') || e.target.closest('.like-btn') || e.target.closest('.user-info')) return;
    const card = e.currentTarget;
    const type = card.getAttribute('data-type');
    const id = parseInt(card.getAttribute('data-id'));
    if (type && id) showDetailPage(type, id);
  }
  function likeHandler(e) {
    e.stopPropagation();
    const btn = e.currentTarget;
    const icon = btn.querySelector('i');
    const countSpan = btn.querySelector('.like-count');
    let count = parseInt(countSpan.innerText);
    if (icon.classList.contains('far')) {
      icon.classList.replace('far', 'fas');
      count++;
      countSpan.innerText = count;
      alert('Liked');
    } else {
      icon.classList.replace('fas', 'far');
      count--;
      countSpan.innerText = count;
      alert('Unliked');
    }
  }
