-- =============================================
-- 10. DUA REPORTS - uuid v7
-- =============================================

CREATE TABLE dua_reports (
    id          UUID PRIMARY KEY DEFAULT uuid_generate_v7(),

    dua_id      UUID NOT NULL REFERENCES duas(id)  ON DELETE CASCADE,
    user_id     UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,

    reason      report_reason NOT NULL,
    description TEXT,

    status      report_status NOT NULL DEFAULT 'pending',
    resolved_at TIMESTAMP,
    resolved_by UUID REFERENCES users(id) ON DELETE SET NULL,

    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    UNIQUE (dua_id, user_id, reason)
);


-- 16. POEM REPORTS - uuid v7
-- =============================================

CREATE TABLE poem_reports (
    id          UUID PRIMARY KEY DEFAULT uuid_generate_v7(),

    poem_id     UUID NOT NULL REFERENCES poems(id)  ON DELETE CASCADE,
    user_id     UUID NOT NULL REFERENCES users(id)  ON DELETE CASCADE,

    reason      poem_report_reason NOT NULL,
    description TEXT,

    status      report_status NOT NULL DEFAULT 'pending',
    resolved_at TIMESTAMP,
    resolved_by UUID REFERENCES users(id) ON DELETE SET NULL,

    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    UNIQUE (poem_id, user_id, reason)
);


CREATE TYPE report_status AS ENUM (
    'pending',
    'resolved',
    'dismissed'
);

CREATE TYPE report_reason AS ENUM (
    'wrong_arabic_text',
    'wrong_transliteration',
    'wrong_translation',
    'wrong_source',
    'inappropriate_content',
    'duplicate_dua',
    'other'
);

CREATE TYPE poem_report_reason AS ENUM (
    'wrong_transliteration',
    'wrong_translation',
    'wrong_author',
    'inappropriate_content',
    'duplicate_poem',
    'copyright_violation',
    'other'
);
